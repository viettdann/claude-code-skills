# Hardcoded API Keys and Secrets
# Usage: rg -i -f scripts/patterns/secrets.txt

# Generic secret patterns
api_key\s*[=:]\s*['"][^'"]{10,}
secret\s*[=:]\s*['"][^'"]{10,}
password\s*[=:]\s*['"][^'"]+
token\s*[=:]\s*['"][^'"]{20,}
private_key\s*[=:]

# Stripe
sk_live_[0-9a-zA-Z]{24,}
sk_test_[0-9a-zA-Z]{24,}
pk_live_[0-9a-zA-Z]{24,}

# AWS
AKIA[0-9A-Z]{16}
aws_secret_access_key\s*[=:]\s*['"][^'"]+

# OpenAI
sk-[a-zA-Z0-9]{48}

# GitHub
ghp_[0-9a-zA-Z]{36}
gho_[0-9a-zA-Z]{36}

# Google
AIza[0-9A-Za-z\-_]{35}

# JWT (very long base64 strings that look like tokens)
eyJ[A-Za-z0-9_-]{30,}

# Database connection strings with credentials
postgres://[^:]+:[^@]+@
mysql://[^:]+:[^@]+@
mongodb://[^:]+:[^@]+@

# Generic long hex strings (potential secrets)
[0-9a-fA-F]{32,}
